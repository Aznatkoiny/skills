% ---- Skill tag ----------------------------------------------
\newcommand{\skilltag}[1]{%
  \tikz[baseline]{
    \node[
      fill=themeTag!40, rounded corners=3pt,
      inner xsep=5pt, inner ysep=2pt,
      font=\tiny\color{themeColor}
    ] {#1};
  }\hspace{1pt}%
}

% ---- Skill rating (filled dots) -----------------------------
% Usage: \skillrating{SkillName}{4}{5}
\newcommand{\skillrating}[3]{%
  \vspace{2pt}
  \begin{tikzpicture}[baseline]
    \node[anchor=west, font=\small\color{sideText}] at (0,0) {#1};
    \foreach \i in {1,...,#3} {
      \pgfmathsetmacro{\filled}{ifthenelse(\i<=#2,1,0)}
      \ifnum\i>#2
        \fill[sideText!30] ({0.16*\i + 2.6}, 0) circle (3.5pt);
      \else
        \fill[themeLight] ({0.16*\i + 2.6}, 0) circle (3.5pt);
      \fi
    }
  \end{tikzpicture}\\
}
